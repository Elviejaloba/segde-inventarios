from __future__ import annotations
import json, os
from typing import Tuple
import firebase_admin
from firebase_admin import credentials, firestore, storage

_app = None
_db = None
_bucket = None

def init_firebase():
    global _app, _db, _bucket
    if _app: 
        return _db, _bucket
    creds_json = os.environ.get("FIREBASE_CREDENTIALS_JSON")
    project_id = os.environ.get("FIREBASE_PROJECT_ID")
    bucket_name = os.environ.get("FIREBASE_STORAGE_BUCKET")
    if not creds_json or not project_id or not bucket_name:
        raise RuntimeError("Missing Firebase secrets: FIREBASE_CREDENTIALS_JSON / PROJECT_ID / STORAGE_BUCKET")
    cred = credentials.Certificate(json.loads(creds_json))
    _app = firebase_admin.initialize_app(cred, {"storageBucket": bucket_name, "projectId": project_id})
    _db = firestore.client()
    _bucket = storage.bucket()
    return _db, _bucket

def upload_to_storage(local_path: str, dest_path: str) -> str:
    _, bucket = init_firebase()
    blob = bucket.blob(dest_path)
    blob.upload_from_filename(local_path, content_type="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet")
    blob.make_public()
    return blob.public_url

def save_inventory_metadata(meta: dict, rows_sample: list[dict] | None = None) -> str:
    db, _ = init_firebase()
    doc_ref = db.collection("tomas_inventario").add({**meta, "created_at": firestore.SERVER_TIMESTAMP, "rows_sample": rows_sample or []})
    return str(doc_ref[1].id)

from __future__ import annotations
import pandas as pd
from typing import Dict, Any, Tuple

DEF_FISICO_COL_IDX = 8   # 0-based => column I
DEF_TEORICO_COL_IDX = 9  # 0-based => column J
DEF_DIF_COL_IDX = 11     # 0-based => column L

POSSIBLE_SUC_HEADERS = ["sucursal","sucursal/deposito","sucursal/depósito","deposito","depósito"]
POSSIBLE_COMP_HEADERS = ["comprobante","nº comprobante","numero de comprobante","número de comprobante","nro comprobante","nro. comprobante"]

def read_excel_preserve(path: str) -> pd.DataFrame:
    # Read without forcing dtype; preserve headers if present
    return pd.read_excel(path, engine="openpyxl", header=0)

def detect_col_by_headers(df: pd.DataFrame, candidates: list[str]) -> int | None:
    if df is None or df.empty: return None
    cols = [str(c).strip().lower() for c in df.columns]
    for idx, name in enumerate(cols):
        for c in candidates:
            if c == name:
                return idx
    return None

def compute_metrics(df: pd.DataFrame, fisico_idx=DEF_FISICO_COL_IDX, teorico_idx=DEF_TEORICO_COL_IDX, dif_idx=DEF_DIF_COL_IDX,
                    suc_idx: int | None = None, comp_idx: int | None = None) -> Dict[str, Any]:
    # Safe access by position (fallback if headers missing)
    df_calc = df.copy()
    ncols = df_calc.shape[1]
    def colsafe(i): return df_calc.iloc[:, i] if 0 <= i < ncols else None

    # Try to detect sucursal/comprobante by header if not provided
    if suc_idx is None:
        sdet = detect_col_by_headers(df_calc, POSSIBLE_SUC_HEADERS)
        if sdet is not None: suc_idx = sdet
    if comp_idx is None:
        cdet = detect_col_by_headers(df_calc, POSSIBLE_COMP_HEADERS)
        if cdet is not None: comp_idx = cdet

    fis = colsafe(fisico_idx)
    teo = colsafe(teorico_idx)
    dif = colsafe(dif_idx)

    # Create a normalized table for aggregation (best-effort numeric)
    out = df_calc.copy()
    if fis is not None: out["__fisico"] = pd.to_numeric(fis, errors="coerce")
    if teo is not None: out["__teorico"] = pd.to_numeric(teo, errors="coerce")
    if dif is not None: out["__diferencia"] = pd.to_numeric(dif, errors="coerce")
    if suc_idx is not None: out["__sucursal"] = df_calc.iloc[:, suc_idx].astype(str)
    if comp_idx is not None: out["__comprobante"] = df_calc.iloc[:, comp_idx].astype(str)

    # Aggregations
    group_keys = []
    if "__sucursal" in out.columns: group_keys.append("__sucursal")
    if "__comprobante" in out.columns: group_keys.append("__comprobante")

    agg = {}
    for k in ["__fisico","__teorico","__diferencia"]:
        if k in out.columns: agg[k] = "sum"
    summary = out.groupby(group_keys).agg(agg).reset_index() if group_keys else out[["__fisico","__teorico","__diferencia"]].sum().to_frame().T

    # Convert previews (limit rows for sample storage)
    preview = df_calc.head(200).fillna("").astype(str).to_dict(orient="records")
    return {
        "detected": {"sucursal_idx": suc_idx, "comprobante_idx": comp_idx},
        "preview_records": preview,
        "summary_records": summary.fillna(0).to_dict(orient="records")
    }

async function postForm(url, formData) {
  const res = await fetch(url, { method: "POST", body: formData });
  if (!res.ok) throw new Error(await res.text());
  return await res.json();
}

function renderTable(container, rows, title) {
  if (!rows || !rows.length) return;
  const cols = Object.keys(rows[0]);
  const table = document.createElement("table");
  table.className = "min-w-full border bg-white rounded-xl overflow-hidden";
  const thead = document.createElement("thead");
  const trh = document.createElement("tr");
  cols.forEach(c => {
    const th = document.createElement("th");
    th.className = "text-left text-xs font-semibold px-3 py-2 border-b";
    th.textContent = c;
    trh.appendChild(th);
  });
  thead.appendChild(trh);
  table.appendChild(thead);
  const tbody = document.createElement("tbody");
  rows.forEach(r => {
    const tr = document.createElement("tr");
    cols.forEach(c => {
      const td = document.createElement("td");
      td.className = "text-xs px-3 py-1 border-b";
      td.textContent = r[c] === null || r[c] === undefined ? "" : String(r[c]);
      tr.appendChild(td);
    });
    tbody.appendChild(tr);
  });
  table.appendChild(tbody);

  const wrap = document.createElement("div");
  if (title) {
    const h = document.createElement("h2");
    h.className = "text-lg font-semibold mb-2";
    h.textContent = title;
    wrap.appendChild(h);
  }
  wrap.appendChild(table);
  container.appendChild(wrap);
}

document.getElementById("uploadForm").addEventListener("submit", async (e) => {
  e.preventDefault();
  const form = e.currentTarget;
  const fd = new FormData(form);
  const results = document.getElementById("results");
  results.innerHTML = "";
  try {
    const data = await postForm("/importacion/upload", fd);
    // Preview original (as-is)
    renderTable(results, data.preview.preview_records, "Vista previa (primeras 200 filas)");
    // Summary
    renderTable(results, data.summary.summary_records, "Resumen por Sucursal / Comprobante");
    // Meta
    const meta = document.createElement("pre");
    meta.className = "text-xs text-gray-600";
    meta.textContent = JSON.stringify({ detected: data.preview.detected, document_id: data.document_id, file_url: data.file_url }, null, 2);
    results.appendChild(meta);
  } catch (err) {
    const p = document.createElement("p");
    p.className = "text-red-600";
    p.textContent = "Error: " + err.message;
    results.appendChild(p);
  }
});

import os, tempfile, time
from flask import Blueprint, request, render_template, jsonify
from werkzeug.utils import secure_filename
import pandas as pd

from services.firebase import init_firebase, upload_to_storage, save_inventory_metadata
from utils.excel_parse import read_excel_preserve, compute_metrics

bp = Blueprint("importacion", __name__, url_prefix="/importacion")

@bp.route("/", methods=["GET"])
def page():
    return render_template("importacion.html")

@bp.route("/upload", methods=["POST"])
def upload():
    sucursal = request.form.get("sucursal", "").strip()
    fecha = request.form.get("fecha", "").strip()
    suc_idx = request.form.get("suc_idx", "").strip()
    comp_idx = request.form.get("comp_idx", "").strip()

    file = request.files.get("file")
    if not file or not file.filename.lower().endswith(".xlsx"):
        return ("Solo se aceptan archivos .xlsx", 400)

    filename = secure_filename(file.filename)
    with tempfile.TemporaryDirectory() as td:
        local_path = os.path.join(td, filename)
        file.save(local_path)
        df = read_excel_preserve(local_path)

        # Compute metrics
        s_idx = int(suc_idx) if suc_idx.isdigit() else None
        c_idx = int(comp_idx) if comp_idx.isdigit() else None
        metrics = compute_metrics(df, suc_idx=s_idx, comp_idx=c_idx)

        # Upload file to Firebase Storage
        ts = int(time.time())
        dest = f"tomas_inventario/{sucursal}/{fecha}/{ts}_{filename}"
        file_url = upload_to_storage(local_path, dest)

    # Persist metadata + sample
    meta = {
        "sucursal": sucursal,
        "fecha": fecha,
        "file_name": filename,
        "file_path": dest,
        "file_url": file_url,
        "detected": metrics["detected"]
    }
    doc_id = save_inventory_metadata(meta, rows_sample=metrics["preview_records"][:20])

    return jsonify({
        "ok": True,
        "document_id": doc_id,
        "file_url": file_url,
        "preview": {
            "detected": metrics["detected"],
            "preview_records": metrics["preview_records"]
        },
        "summary": {
            "summary_records": metrics["summary_records"]
        }
    })

import os, tempfile, time
from fastapi import APIRouter, UploadFile, Form, HTTPException
from fastapi.responses import HTMLResponse, JSONResponse
from fastapi import Request
from starlette.templating import Jinja2Templates

from services.firebase import init_firebase, upload_to_storage, save_inventory_metadata
from utils.excel_parse import read_excel_preserve, compute_metrics

router = APIRouter(prefix="/importacion", tags=["importacion"])
templates = Jinja2Templates(directory="templates")

@router.get("/", response_class=HTMLResponse)
async def page(request: Request):
    return templates.TemplateResponse("importacion.html", {"request": request})

@router.post("/upload")
async def upload(
    file: UploadFile,
    sucursal: str = Form(...),
    fecha: str = Form(...),
    suc_idx: str = Form(""),
    comp_idx: str = Form("")
):
    if not file.filename.lower().endswith(".xlsx"):
        raise HTTPException(status_code=400, detail="Solo .xlsx")

    filename = file.filename
    contents = await file.read()
    with tempfile.TemporaryDirectory() as td:
        local_path = os.path.join(td, filename)
        with open(local_path, "wb") as f:
            f.write(contents)

        df = read_excel_preserve(local_path)
        s_idx = int(suc_idx) if suc_idx.isdigit() else None
        c_idx = int(comp_idx) if comp_idx.isdigit() else None
        metrics = compute_metrics(df, suc_idx=s_idx, comp_idx=c_idx)

        ts = int(time.time())
        dest = f"tomas_inventario/{sucursal}/{fecha}/{ts}_{filename}"
        file_url = upload_to_storage(local_path, dest)

    meta = {
        "sucursal": sucursal,
        "fecha": fecha,
        "file_name": filename,
        "file_path": dest,
        "file_url": file_url,
        "detected": metrics["detected"]
    }
    doc_id = save_inventory_metadata(meta, rows_sample=metrics["preview_records"][:20])

    return JSONResponse({
        "ok": True,
        "document_id": doc_id,
        "file_url": file_url,
        "preview": {
            "detected": metrics["detected"],
            "preview_records": metrics["preview_records"]
        },
        "summary": {
            "summary_records": metrics["summary_records"]
        }
    })
